      def search(*args)
        options = args.last.is_a?(Hash) ? args.pop : {}
        client = Riddle::Client.new
        client.match_mode = :extended
        q = client.query("#{args} @class #{name}")
        ids = q[:matches].map {|r| r[:doc]}
        find(ids)
      end
