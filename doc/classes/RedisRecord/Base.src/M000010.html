      def populate_from_yaml(yaml_file)
        entries = YAML::load_file(yaml_file)
        records = []
        entries.keys.each do |k|
          begin
            r = new(entries[k])
                  r.save 
                  records << r
          rescue Exception => e 
                  records << nil
          end
        end
        records
      end
