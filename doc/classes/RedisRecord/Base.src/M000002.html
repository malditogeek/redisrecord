      def find(*args)
        options = args.last.is_a?(Hash) ? args.pop : {} 
        records = []
        case args.first
          when :all
            find_all(options)
          when :first
            find_first(options)
          when :last
            find_last(options)
          else
            find_from_ids(args, options)
        end
      end
