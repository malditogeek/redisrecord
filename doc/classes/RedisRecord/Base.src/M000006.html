    def destroy
      @@redis.list_rm("#{self.class.name}:list", 0, @cached_attrs[:id])
      @cached_attrs.keys.each do |k|
        @@redis.delete("#{self.class.name}:#{@cached_attrs[:id]}:#{k}")
      end
      @@redis.delete("#{self.class.name}:#{id}:attrs")
    end
