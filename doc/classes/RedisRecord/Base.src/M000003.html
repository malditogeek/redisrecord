      def sort_by(attribute, options={})
        limit  = options[:limit]  || 10
        offset = options[:offset] || 0
        order  = options[:order]  || 'ALPHA DESC'
        ids = @@redis.sort("#{name}:list", 
                           :by => "#{name}:*:#{attribute}", 
                           :limit => [offset, limit],
                           :order => order,
                           :get => "#{name}:*:id")
        find_from_ids(ids, options)
      end
